sigma = 1.0
x_min = mu - 5*sigma
x_max = mu + 5*sigma

# Sinh N điểm x ngẫu nhiên trong khoảng [x_min, x_max] sử dụng np.random.rand()
x = np.random.rand(N) * (x_max - x_min) + x_min  # [0,1) * (b-a) + a --> [a, b)

""" CÁC BẠN LÀM BÀI Ở ĐÂY """
print("\n4. Giải hệ phương trình tuyến tính")

# Hệ phương trình:
# 4x1 + 3x2 - 5x3 = 2
# -2x1 - 4x2 + 5x3 = 5
# 8x1 + 8x2 = -3

# Ma trận hệ số A
A = np.array([[4, 3, -5],
              [-2, -4, 5],
              [8, 8, 0]])

# Vector kết quả y
y = np.array([2, 5, -3])

# Giải hệ phương trình Ax = y
x = np.linalg.solve(A, y)

Bài 3. Tính gần đúng số pi dùng phương pháp Monte Carlo¶

Vẽ một đường tròn $C$ có bán kính $r$ nội tiếp một hình vuông $C$, và đặt ngẫu nhiên $N$ chấm lên hình vuông. Tỉ lệ các chấm nằm trong hình tròn trên tổng số chấm xấp xỉ bằng diện tích của hình tròn chia cho hình vuông như sau: $$ \frac{N_C}{N} \approx \frac{\mathbb{S}_C}{\mathbb{S}_N} $$ nơi mà $N_C$ là các chấm trong hình tròn, $N$ là tổng số chấm, $\mathbb{S}_C$ và $\mathbb{S}_N$ lần lượt là diện tích của hình tròn và hình vuông. $$ \mathbb{S}_C = \pi * r^2 $$ $$ \mathbb{S}_N = (2*r)^2 = 4r^2 $$ Do đó, công thức trên trở thành: $$ \frac{N_C}{N} \approx \frac{\pi * r^2}{4r^2} = \frac{\pi}{4} $$ Suy ra, ta có giá trị số $\pi$ được tính như sau: $$ \pi \approx \frac{4N_C}{N} $$

Các bạn hãy lập trình tính sấp xỉ số $\pi$ dùng phương pháp Monte Carlo. Cho biết các sai số với $N = 100$, $N = 10000$, và $N = 1000000$. Vẽ đồ thị minh họa cho chương trình.

Hướng dẫn: Viết chương trình phát sinh $N$ điểm $(x_i, y_i)$ với $x_i, y_i \in [-1, -1]$. $(x_i, y_i)$ thuộc hình tròn tâm $(0, 0)$ bán kính 1 nội tiếp hình vuông có tâm tại $(0, 0)$ độ dài cạnh 2 khi và chỉ khi $x_i^2 + y_i^2 <= 1$.

import numpy as np

def calc_pi_monte_carlo(n = 100):
    pi = 0
    
    # gợi ý: sử dụng np.random.rand(s1, s2, ...) --> sinh ngẫu nhiên các số trong nửa đoạn [0.0, 1.0)
    """ CÁC BẠN LÀM BÀI Ở ĐÂY """    
    # Sinh ngẫu nhiên N điểm (x, y) trong khoảng [0, 1) sử dụng np.random.rand()
    # Sau đó chuyển đổi sang khoảng [-1, 1] bằng công thức: 2*x - 1
    x = 2 * np.random.rand(n) - 1  # Chuyển [0,1) -> [-1,1)
    y = 2 * np.random.rand(n) - 1  # Chuyển [0,1) -> [-1,1)
    
    # Tính khoảng cách từ mỗi điểm đến tâm (0,0)
    distances = x**2 + y**2
    
    # Đếm số điểm nằm trong hình tròn (khoảng cách < 1)
    points_inside = np.sum(distances < 1)
    
    # Tính xấp xỉ số pi
    pi = 4 * points_inside / n
    
    return pi

#Vẽ đồ thị với N = 100
def plot_monte_carlo(n=100):
    # Sinh điểm sử dụng np.random.rand() theo gợi ý
    x = 2 * np.random.rand(n) - 1  # Chuyển [0,1) -> [-1,1)
    y = 2 * np.random.rand(n) - 1  # Chuyển [0,1) -> [-1,1)
    
    distances = x**2 + y**2
    
    # Phân loại điểm trong và ngoài hình tròn
    inside = distances < 1
    outside = ~inside
    
    # Tính π xấp xỉ
    pi_approx = 4 * np.sum(inside) / n
    
    # Vẽ đồ thị
    fig, ax = plt.subplots(figsize=(6, 6))
    
    # Vẽ hình tròn
    circle = plt.Circle((0, 0), 1, color='g', fill=False, linewidth=2)
    ax.add_patch(circle)
    
    # Vẽ điểm trong hình tròn (màu xanh)
    plt.scatter(x[inside], y[inside], color='blue', s=5, alpha=0.6, label=f'Inside ({np.sum(inside)})')
    
    # Vẽ điểm ngoài hình tròn (màu đỏ)
    plt.scatter(x[outside], y[outside], color='red', s=5, alpha=0.6, label=f'Outside ({np.sum(outside)})')
    
    # Vẽ điểm tại tâm
    plt.plot(0, 0, 'o', color='black', markersize=5)
    
    plt.xlim(-1, 1)
    plt.ylim(-1, 1)
    plt.title(f'Monte Carlo Simulation (N={n})\nπ ≈ {pi_approx:.4f}\nSai số: {abs(pi_approx - math.pi):.6f}')
    plt.legend()
    plt.grid(True)
    plt.axis('equal')
    plt.savefig(f'pi_monte_carlo_N_{n}.png')
    plt.show()
    
    return pi_approx
