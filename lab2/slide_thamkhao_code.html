# apply() mt hm khc trn series
def Tang(x):
    return x if x > 500 else x + 1000

chi_so = ["Ke toan", "KT", "CNTT", "Co khi"]
gia_tri = [310, 360, 580, 340]
S = pd.Series(gia_tri, chi_so)
# p dng Tang trn S (khng thay i S)
print(S.apply(Tang))
